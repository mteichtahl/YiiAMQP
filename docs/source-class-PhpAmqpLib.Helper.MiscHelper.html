 43:      */
 44:     public static function hexdump($data, $htmloutput = true, $uppercase = false, $return = false)
 45:     {
 46:         // Init
 47:         $hexi   = '';
 48:         $ascii  = '';
 49:         $dump   = ($htmloutput === true) ? '<pre>' : '';
 50:         $offset = 0;

 55:

 58:             // Convert to hexidecimal
 59:             $hexi .= sprintf("%02$x ", ord($data[$i]));
 60: 
 61:             // Replace non-viewable bytes with '.'
 62:             if (ord($data[$i]) >= 32) {

 66:             } else {
