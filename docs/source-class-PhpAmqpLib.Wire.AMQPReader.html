 70:      * Wait until some data is retrieved from the socket.
 71:      * 
 72:      * AMQPTimeoutException can be raised if the timeout is set
 73:      *
 74:      * @throws \PhpAmqpLib\Exception\AMQPTimeoutException
 75:      */
 76:     protected function wait()
 77:     {
 78:         if ($this->timeout == 0 && $this->sock) {
 79:             return;
 80:         }
 81: 
 82:         $read   = array($this->sock->real_sock());
 83:         $write  = null;
 84:         $except = null;
 85: 
 86:         // wait ..
 87:         $result = stream_select($read, $write, $except, $this->timeout);
 88: 
 89:         if ($result === false) {
 90:             throw new AMQPRuntimeException(sprintf("An error occurs", $this->timeout));
 91:         }
 92: 
 93:         if ($result === 0) {
 94:             throw new AMQPTimeoutException(sprintf("A timeout occurs while waiting for incoming data", $this->timeout));
 95:         }
 96:     }

 98:     /**
 99:      * @param $n
100:      *
101:      * @return string
102:      * @throws \RuntimeException
103:      * @throws \PhpAmqpLib\Exception\AMQPRuntimeException
104:      */
105:     private function rawread($n)
106:     {
107:         if ($this->sock) {
108:             $res = '';
109:             $read = 0;
110: 
111:             while (true) {
112:                 $this->wait();
113: 
114:                 $buf = fread($this->sock->real_sock(), $n - $read);
115: 
116:                 if ($buf !== false) {
117:                     $read += strlen($buf);
118:                     $res .= $buf;
119:                 }
120: 
121:                 if (feof($this->sock->real_sock()) || ($n - $read) === 0) {
122:                     break;
123:                 }
124:             }
125: 
126:             if (strlen($res) != $n) {
127:                 throw new AMQPRuntimeException("Error reading data. Received " .
128:                     strlen($res) . " instead of expected $n bytes");
129:             }
130: 
131:             $this->offset += $n;
132:         } else {
133:             if (strlen($this->str) < $n) {
134:                 throw new AMQPRuntimeException("Error reading data. Requested $n bytes while string buffer has only " .
135:                     strlen($this->str));
136:             }
137: 
138:             $res = substr($this->str,0,$n);
139:             $this->str = substr($this->str,$n);
140:             $this->offset += $n;
141:         }
142: 
143:         return $res;
144:     }
